# 预测导向压缩系统实验路线说明书

## 一、实验路线总览

本实验路线基于以下核心决策制定：

| 决策项 | 选择 | 核心理由 |
|--------|------|----------|
| 压缩形式 | 预测导向压缩 | 压缩目标不是重构，而是预测能力 |
| 世界模型 | 先实现P(s' | 从"被动观察"转向"主动干预" |
| 探索机制 | 渐进式 | ε-贪心 → 临界 → 不确定性 |
| 验证层次 | L2为目标 | 检验"压缩→预测"框架有效性 |
| 系统定位 | 资源高效自主智能 | 边缘智能、实时系统、终身学习 |

### 阶段划分

本实验路线分为三个阶段：

**第一阶段（1-2周）：预测导向压缩基础**

目标：实现基础的"压缩→预测"循环，验证预测机制的有效性

**第二阶段（2-4周）：状态转移模型**

目标：实现P(s'|s,a)，使系统具备"主动干预"能力

**第三阶段（4-8周）：完整智能循环**

目标：整合探索机制、价值函数，形成完整的自主智能系统

## 二、第一阶段：预测导向压缩基础

### 2.1 核心目标

实现基础的"压缩→预测"循环，证明预测机制比随机机制更有效。

### 2.2 关键代码实现

#### 2.2.1 预测导向压缩模块

```python
import numpy as np
from typing import List, Tuple, Optional


class PredictiveCompressor:
    """
    预测导向压缩器

    核心思想：压缩的目标不是重构输入，而是预测未来
    压缩的质量不是用"重构误差"衡量，而是用"预测误差"衡量
    """

    def __init__(self, input_dim: int, compress_dim: int, learning_rate: float = 0.01):
        self.input_dim = input_dim
        self.compress_dim = compress_dim
        self.lr = learning_rate

        # 编码器：输入 -> 压缩表示
        # 使用简单的线性投影 + ReLU激活
        self.encoder_weights = np.random.randn(input_dim, compress_dim) * 0.1
        self.encoder_bias = np.zeros(compress_dim)

        # 预测器：当前压缩表示 -> 下一时刻压缩表示
        self.predictor_weights = np.random.randn(compress_dim, compress_dim) * 0.1
        self.predictor_bias = np.zeros(compress_dim)

        # 历史记录
        self.compressed_history = []
        self.prediction_errors = []

    def compress(self, x: np.ndarray) -> np.ndarray:
        """
        压缩：将高维输入压缩为低维表示

        参数：
            x: 原始输入，形状 (input_dim,)

        返回：
            压缩表示，形状 (compress_dim,)
        """
        # 线性投影
        compressed = x @ self.encoder_weights + self.encoder_bias
        # ReLU激活
        compressed = np.maximum(0, compressed)
        # L2归一化（可选，增加稳定性）
        norm = np.linalg.norm(compressed)
        if norm > 0:
            compressed = compressed / norm
        return compressed

    def predict(self, compressed: np.ndarray) -> np.ndarray:
        """
        预测：基于当前压缩表示，预测下一时刻的压缩表示

        参数：
            compressed: 当前压缩表示

        返回：
            预测的下一时刻压缩表示
        """
        predicted = compressed @ self.predictor_weights + self.predictor_bias
        predicted = np.maximum(0, predicted)  # ReLU
        return predicted

    def update(self, compressed_curr: np.ndarray, compressed_next: np.ndarray):
        """
        更新预测器参数，基于预测误差

        参数：
            compressed_curr: 当前压缩表示
            compressed_next: 真实的下一时刻压缩表示
        """
        # 预测
        predicted = self.predict(compressed_curr)

        # 计算预测误差
        error = compressed_next - predicted
        self.prediction_errors.append(np.linalg.norm(error))

        # 预测器梯度更新（简化版：直接用误差调整权重）
        # 简化实现：predicted = W * compressed_curr + b
        # 梯度：d_error/dW = -2 * error * compressed_curr
        self.predictor_weights -= self.lr * np.outer(compressed_curr, error)
        self.predictor_bias -= self.lr * error

    def evaluate_compression_quality(self) -> dict:
        """
        评估压缩质量

        返回：
            包含各种评估指标的字典
        """
        if len(self.prediction_errors) == 0:
            return {"mean_error": 0, "variance": 0, "trend": 0}

        errors = np.array(self.prediction_errors[-100:])  # 最近100个
        return {
            "mean_error": float(np.mean(errors)),
            "variance": float(np.var(errors)),
            "trend": float(np.mean(errors[-10:]) - np.mean(errors[:10])) if len(errors) >= 20 else 0
        }


class RandomBaseline:
    """
    随机基线：用于对比

    随机选择压缩表示，不做任何预测
    """

    def __init__(self, compress_dim: int):
        self.compress_dim = compress_dim
        self.compressed_history = []

    def compress(self, x: np.ndarray) -> np.ndarray:
        # 随机生成压缩表示
        compressed = np.random.randn(self.compress_dim)
        compressed = compressed / (np.linalg.norm(compressed) + 1e-8)
        return compressed

    def predict(self, compressed: np.ndarray) -> np.ndarray:
        # 随机预测
        return np.random.randn(self.compress_dim)
```

#### 2.2.2 预测驱动表征选择器

```python
class PredictiveSelector:
    """
    预测驱动表征选择器

    核心思想：选择"预测误差最小"的表征，而非"与当前输入最匹配"的表征
    这是"前瞻性选择"vs"后顾性选择"的区别
    """

    def __init__(self, representations: List[np.ndarray], predictor):
        self.representations = representations  # 表征列表
        self.predictor = predictor  # 预测器
        self.prediction_scores = {}  # 每个表征的预测得分

    def compute_prediction_scores(self, current_input: np.ndarray) -> dict:
        """
        计算每个表征的"预测得分"

        预测得分 = 基于该表征预测下一输入的误差
        误差越小，得分越高
        """
        scores = {}

        for i, rep in enumerate(self.representations):
            # 使用表征压缩当前输入
            compressed = self.predictor.compress(current_input)

            # 使用预测器预测下一时刻
            predicted_next = self.predictor.predict(compressed)

            # 预测得分 = -预测误差（误差越小，得分越高）
            # 这里简化：直接用表征与输入的相似度作为预测得分代理
            score = np.dot(rep, current_input) / (np.linalg.norm(rep) + 1e-8)
            scores[i] = score

        self.prediction_scores = scores
        return scores

    def select(self, current_input: np.ndarray, exploration_rate: float = 0.1) -> int:
        """
        选择表征

        参数：
            current_input: 当前输入
            exploration_rate: 探索概率

        返回：
            选中的表征索引
        """
        # ε-贪心策略
        if np.random.random() < exploration_rate:
            # 探索：随机选择
            return np.random.randint(len(self.representations))

        # 计算预测得分
        scores = self.compute_prediction_scores(current_input)

        # 选择得分最高的表征
        best_idx = max(scores, key=scores.get)
        return best_idx
```

#### 2.2.3 完整系统集成

```python
class PredictiveFCRS:
    """
    预测导向有限竞争表征系统

    整合：压缩 → 预测 → 选择 → 反馈 → 更新
    """

    def __init__(self,
                 input_dim: int = 10,
                 compress_dim: int = 5,
                 pool_capacity: int = 10,
                 learning_rate: float = 0.01,
                 exploration_rate: float = 0.1):

        self.input_dim = input_dim
        self.compress_dim = compress_dim
        self.pool_capacity = pool_capacity
        self.exploration_rate = exploration_rate

        # 核心组件
        self.compressor = PredictiveCompressor(input_dim, compress_dim, learning_rate)
        self.representations = []
        self.selector = None  # 初始化后设置

        # 统计
        self.step_count = 0
        self.prediction_error_history = []
        self.reconstruction_error_history = []

        # 初始化表征池
        self._init_representations()

    def _init_representations(self):
        """初始化表征池"""
        for _ in range(3):
            rep = np.random.randn(self.input_dim)
            rep = rep / (np.linalg.norm(rep) + 1e-8)
            self.representations.append(rep)

        self.selector = PredictiveSelector(self.representations, self.compressor)

    def step(self, x: np.ndarray) -> dict:
        """
        执行一步

        参数：
            x: 输入向量

        返回：
            包含各种误差的字典
        """
        self.step_count += 1

        # 1. 压缩
        compressed = self.compressor.compress(x)
        self.compressor.compressed_history.append(compressed)

        # 2. 预测
        if len(self.compressor.compressed_history) >= 2:
            compressed_prev = self.compressor.compressed_history[-2]
            predicted = self.compressor.predict(compressed_prev)

            # 计算预测误差
            pred_error = np.linalg.norm(predicted - compressed)
            self.prediction_error_history.append(pred_error)

            # 更新预测器
            self.compressor.update(compressed_prev, compressed)

        # 3. 选择表征
        selected_idx = self.selector.select(x, self.exploration_rate)
        selected_rep = self.representations[selected_idx]

        # 4. 计算重构误差（用于对比）
        recon_error = np.linalg.norm(x - selected_rep)
        self.reconstruction_error_history.append(recon_error)

        # 5. 更新表征（简单在线学习）
        self.representations[selected_idx] += 0.01 * (x - selected_rep)
        # 归一化
        self.representations[selected_idx] /= (np.linalg.norm(self.representations[selected_idx]) + 1e-8)

        return {
            "step": self.step_count,
            "prediction_error": self.prediction_error_history[-1] if self.prediction_error_history else None,
            "reconstruction_error": recon_error,
            "pool_size": len(self.representations)
        }

    def run(self, env, steps: int) -> dict:
        """
        运行多步

        参数：
            env: 环境（需要提供generate_input()方法）
            steps: 步数

        返回：
            最终统计
        """
        for _ in range(steps):
            x = env.generate_input()
            self.step(x)

        return self.get_statistics()

    def get_statistics(self) -> dict:
        """获取统计信息"""
        pred_errors = np.array(self.prediction_error_history[-100:]) if self.prediction_error_history else []
        recon_errors = np.array(self.reconstruction_error_history[-100:]) if self.reconstruction_error_history else []

        return {
            "step": self.step_count,
            "pool_size": len(self.representations),
            "mean_prediction_error": float(np.mean(pred_errors)) if len(pred_errors) > 0 else None,
            "mean_reconstruction_error": float(np.mean(recon_errors)) if len(recon_errors) > 0 else None,
            "compression_ratio": self.input_dim / self.compress_dim
        }
```

## 三、第二阶段：状态转移模型

### 3.1 核心目标

实现状态转移函数P(s'|s,a)，使系统从"被动观察"转向"主动干预"。

### 3.2 关键代码实现

#### 3.2.1 世界模型模块

```python
class WorldModel:
    """
    世界模型

    包含：
    - 观测函数 P(o|s)
    - 状态转移函数 P(s'|s,a)
    - 价值函数 R(s,a) [可选]
    """

    def __init__(self, state_dim: int, action_dim: int, hidden_dim: int = 20):
        self.state_dim = state_dim
        self.action_dim = action_dim
        self.hidden_dim = hidden_dim

        # 观测函数：状态 -> 观测（简化为恒等映射）
        # 在更复杂的系统中，这会是一个神经网络

        # 状态转移函数：状态 + 行动 -> 下一状态
        # 使用简单的线性模型
        self.transition_weights = np.random.randn(state_dim + action_dim, state_dim) * 0.1
        self.transition_bias = np.zeros(state_dim)

        # 转移预测误差历史
        self.transition_errors = []

    def predict_next_state(self, state: np.ndarray, action: np.ndarray) -> np.ndarray:
        """
        预测下一状态

        参数：
            state: 当前状态
            action: 采取的行动

        返回：
            预测的下一状态
        """
        # 拼接状态和行动
        state_action = np.concatenate([state, action])

        # 线性预测 + ReLU
        next_state = state_action @ self.transition_weights + self.transition_bias
        next_state = np.maximum(0, next_state)  # ReLU
        next_state = next_state / (np.linalg.norm(next_state) + 1e-8)  # 归一化

        return next_state

    def update(self, state: np.ndarray, action: np.ndarray, actual_next_state: np.ndarray):
        """
        更新状态转移模型

        参数：
            state: 当前状态
            action: 采取的行动
            actual_next_state: 实际的下一状态
        """
        predicted = self.predict_next_state(state, action)
        error = actual_next_state - predicted

        self.transition_errors.append(np.linalg.norm(error))

        # 简化更新
        state_action = np.concatenate([state, action])
        self.transition_weights -= 0.01 * np.outer(state_action, error)
        self.transition_bias -= 0.01 * error

    def imagine_rollout(self, initial_state: np.ndarray, actions: List[np.ndarray]) -> List[np.ndarray]:
        """
        心理模拟：想象执行一系列行动后的状态序列

        参数：
            initial_state: 初始状态
            actions: 行动序列

        返回：
            想象的状态序列
        """
        states = [initial_state]

        for action in actions:
            next_state = self.predict_next_state(states[-1], action)
            states.append(next_state)

        return states
```

#### 3.2.2 主动规划器

```python
class ActivePlanner:
    """
    主动规划器

    使用世界模型进行"心理模拟"，选择最优行动序列
    """

    def __init__(self, world_model: WorldModel, planning_horizon: int = 3):
        self.world_model = world_model
        self.planning_horizon = planning_horizon

    def plan(self, current_state: np.ndarray, possible_actions: List[np.ndarray]) -> np.ndarray:
        """
        规划：选择最优行动

        参数：
            current_state: 当前状态
            possible_actions: 可能的行动列表

        返回：
            选择的行动
        """
        best_action = None
        best_score = -float('inf')

        for action in possible_actions:
            # 心理模拟
            imagined_states = self.world_model.imagine_rollout(current_state, [action])

            # 评估：简单使用最终状态的范数作为得分
            # 更复杂的实现应该使用价值函数
            score = np.linalg.norm(imagined_states[-1])

            if score > best_score:
                best_score = score
                best_action = action

        return best_action if best_action is not None else possible_actions[0]
```

```python
class CompleteIntelligentSystem:
    """
    完整智能系统

    包含：
    - 感知层：观察输入
    - 压缩层：提取特征
    - 预测层：预测未来
    - 行动层：选择行动
    - 反馈层：评估结果
    - 更新层：更新模型
    """

    def __init__(self, input_dim: int, action_dim: int, compress_dim: int):
        self.input_dim = input_dim
        self.action_dim = action_dim
        self.compress_dim = compress_dim

        # 各层组件
        self.compressor = PredictiveCompressor(input_dim, compress_dim)
        self.world_model = WorldModel(compress_dim, action_dim)
        self.planner = ActivePlanner(self.world_model)

        # 历史记录
        self.state_history = []
        self.action_history = []
        self.reward_history = []

    def step(self, observation: np.ndarray, possible_actions: List[np.ndarray]) -> tuple:
        """
        执行一步完整智能循环

        参数：
            observation: 当前观察
            possible_actions: 可能的行动列表

        返回：
            (选择的行动, 预测信息, 实际反馈)
        """
        # 1. 压缩观察
        compressed = self.compressor.compress(observation)
        self.state_history.append(compressed)

        # 2. 预测（如果有历史）
        prediction = None
        if len(self.state_history) >= 2:
            prediction = self.compressor.predict(self.state_history[-2])

        # 3. 选择行动（使用规划器或ε-贪心）
        if np.random.random() < 0.2:  # 探索
            action = possible_actions[np.random.randint(0, len(possible_actions))]
        else:  # 利用
            action = self.planner.plan(compressed, possible_actions)

        self.action_history.append(action)

        return action, prediction, compressed

    def update_from_feedback(self, next_observation: np.ndarray, reward: float):
        """
        从反馈中更新

        参数：
            next_observation: 下一观察
            reward: 奖励
        """
        # 更新压缩器
        if len(self.state_history) >= 2:
            self.compressor.update(self.state_history[-2], self.state_history[-1])

        # 更新世界模型
        if len(self.state_history) >= 1 and len(self.action_history) >= 1:
            self.world_model.update(
                self.state_history[-1],
                self.action_history[-1],
                self.compressor.compress(next_observation)
            )

        # 记录奖励
        self.reward_history.append(reward)
```
